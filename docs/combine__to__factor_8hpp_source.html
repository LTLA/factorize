49 if (ninputs == 0) {

58 // Creating a hashmap on the combinations of each factor.
59 struct Combination {
60 Combination(const std::size_t i) : index(i) {}
61 std::size_t index;
62 };
63
64 auto unique = [&]{ // scoping this in an IIFE to release map memory sooner.
65 // Using a map with a custom comparator that uses the index
66 // of first occurrence of each factor as the key. Currently using a map
67 // to (i) avoid issues with collisions of combined hashes and (ii)
68 // avoid having to write more code for sorting a vector of arrays.
69 auto cmp = [&](const Combination& left, const Combination& right) -> bool {
70 for (auto curf : inputs) {
71 if (curf[left.index] < curf[right.index]) {
72 return true;
73 } else if (curf[left.index] > curf[right.index]) {
74 return false;
75 }
76 }
77 return false;
78 };
79 std::map<Combination, Code_, decltype(I(cmp))> mapping(std::move(cmp));
80
81 const auto eq = [&](const Combination& left, const Combination& right) -> bool {
82 for (auto curf : inputs) {
83 if (curf[left.index] != curf[right.index]) {
84 return false;
85 }
86 }
87 return true;
88 };
89
90 for (decltype(I(n)) i = 0; i < n; ++i) {
91 Combination current(i);
92 const auto mIt = mapping.find(current);
93 if (mIt == mapping.end() || !eq(mIt->first, current)) {
94 Code_ alt = mapping.size();
95 mapping.insert(mIt, std::make_pair(current, alt));
96 codes[i] = alt;
97 } else {
98 codes[i] = mIt->second;
99 }
100 }
101
102 return std::vector<std::pair<Combination, Code_> >(mapping.begin(), mapping.end());
103 }();
104
105 // Remapping to a sorted set.
106 const auto nuniq = unique.size();
107 for (auto& ofac : output) {
108 ofac.reserve(nuniq);
109 }
110 auto remapping = sanisizer::create<std::vector<Code_> >(nuniq);
111 for (decltype(I(nuniq)) u = 0; u < nuniq; ++u) {
112 const auto ix = unique[u].first.index;
113 for (decltype(I(ninputs)) f = 0; f < ninputs; ++f) {
114 output[f].push_back(inputs[f][ix]);
115 }
116 remapping[unique[u].second] = u;
117 }
118
119 // Mapping each cell to its sorted combination.
120 for (decltype(I(n)) i = 0; i < n; ++i) {
121 codes[i] = remapping[codes[i]];
122 }
123
124 return output;
125}

149template<typename Input_, typename Number_, typename Code_>
150std::vector<std::vector<Input_> > combine_to_factor_unused(const std::size_t n, const std::vector<std::pair<const Input_*, Number_> >& inputs, Code_* const codes) {
151 const auto ninputs = inputs.size();
152 auto output = sanisizer::create<std::vector<std::vector<Input_> > >(ninputs);
153
154 // Handling the special cases.
155 if (ninputs == 0) {
156 std::fill_n(codes, n, 0);
157 return output;
158 }
159 if (ninputs == 1) {
160 sanisizer::resize(output[0], inputs[0].second);
161 std::iota(output[0].begin(), output[0].end(), static_cast<Code_>(0));
162 std::copy_n(inputs[0].first, n, codes);
163 return output;
164 }
165
166 // We iterate from back to front, where the first factor is the slowest changing.
167 std::copy_n(inputs[ninputs - 1].first, n, codes);
168 Code_ ncombos = inputs[ninputs - 1].second;
169 for (decltype(I(ninputs)) f = ninputs - 1; f > 0; --f) {
170 const auto& finfo = inputs[f - 1];
171 const auto next_ncombos = sanisizer::product<Code_>(ncombos, finfo.second);
172 const auto ff = finfo.first;
173 for (decltype(I(n)) i = 0; i < n; ++i) {
174 // Product is safe as it is obviously less than 'next_combos' for 'ff[i] < finfo.second'.
175 // Addition is also safe as it will be less than 'next_combos', though this is less obvious.
176 codes[i] += sanisizer::product_unsafe<Code_>(ncombos, ff[i]);
177 }
178 ncombos = next_ncombos;
179 }
180
181 sanisizer::cast<decltype(I(output[0].size()))>(ncombos); // check that we can actually make the output vectors.
182 Code_ outer_repeats = ncombos;
183 Code_ inner_repeats = 1;
184 for (decltype(I(ninputs)) f = ninputs; f > 0; --f) {
185 auto& out = output[f - 1];
186 out.reserve(ncombos);
187
188 const auto& finfo = inputs[f - 1];
189 const Code_ initial_size = inner_repeats * finfo.second;
190 out.resize(initial_size);
191
192 if (inner_repeats == 1) {
193 std::iota(out.begin(), out.end(), static_cast<Code_>(0));
194 } else {
195 auto oIt = out.begin();
196 for (Number_ l = 0; l < finfo.second; ++l) {
197 std::fill_n(oIt, inner_repeats, l);
198 oIt += inner_repeats;
199 }
200 }
201 inner_repeats = initial_size;
202
203 outer_repeats /= finfo.second;
204 for (Code_ r = 1; r < outer_repeats; ++r) {
205 // Referencing to 'begin()' while inserting is safe, as we reserved the full length at the start.
206 // Thus, there shouldn't be any allocations.
207 out.insert(out.end(), out.begin(), out.begin() + initial_size);
208 }
209 }
210
211 return output;
212}
